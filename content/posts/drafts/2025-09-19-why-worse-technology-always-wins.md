```lisp
(defun factorial (n)
  "Calculate the factorial of N."
  (if (<= n 1)
      1
      (* n (factorial (- n 1)))))

(factorial 5) ; => 120
```

And some Python:

```python
def fibonacci(n):
    """Generate the nth Fibonacci number."""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(10))  # Output: 55
```
